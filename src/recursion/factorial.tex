\begin{frame}{Wzory}
    \begin{definition}
        \[ n! = \prod\limits_{k = 1}^{n}k \qquad \text{, for } n \geq 1 \]
        \[ 0! = 1 \]
    \end{definition}
    \begin{definition}[recursion]
        \[ n! =
        \begin{cases}
            1 & \text{, for } n = 0 \\
            n \cdot (n - 1)! & \text{, for } n \geq 1
        \end{cases}
        \]
    \end{definition}
\end{frame}
